import time
starttime = time.time()

num = 0
number = 2520
while (True):
	i = 11
	while (i <= 20):
		if (number % i != 0):
			break
		if (i == 20):
			num = number
		i += 1
	if (num != 0) :
		break
	number += 2520

print str(num)
finishtime = time.time()
print "Executed in " + str(finishtime - starttime) + " seconds."
